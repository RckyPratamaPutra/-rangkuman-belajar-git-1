1. ketik " ssh-keygen -t rsa " untuk generate ke ssh key di github (1x) 2. ketik " cat ~/.ssh/id_rsa.pub " copy ssh key ke github ssh key (1x) 3. membuat folder project ketik " mkdir (nama folder) " 4. masuk ke folder ketik " cd project " 5. buat file .html di sublime 6. ketik " git init " untuk memakai git 6. ketik " git add index,html " untuk mengadd file html ke github 7. ketik " git status " untuk melihat status commit apa blom 8. ketik " git commit " untuk memberikan keterangan file yang di push 9. ketik " git config --global user.email ("masukan email anda") " (1x) 10. KETIK " git config --global user.nama ("masukan nama anda") " (1x) 11. ketik " git remote add origin (masukan ssh url project yg di github) " 12. ketik " git remote -v " untuk melihat akses project 13. ketik " git push origin master " untuk mengupload ke github 14. " git pull origin master " untuk mendowload jeng merge file-file 15. " git checkout -b (nama branch) " untuk menambahkan branch 16. " git checkout (nama branch) " untuk pindah ke branch yang lain
